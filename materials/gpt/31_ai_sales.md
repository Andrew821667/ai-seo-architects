### Automated follow-up sequences
```python
# Пример системы автоматического follow-up
class AutomatedFollowUp:
    def __init__(self, crm_client, email_client):
        self.crm = crm_client
        self.email = email_client
    
    def trigger_sequence(self, lead_id, sequence_type):
        """Запуск последовательности follow-up"""
        lead = self.crm.get_lead(lead_id)
        sequence = self.get_sequence(sequence_type, lead)
        
        for step in sequence:
            self.schedule_action(lead_id, step)
    
    def get_sequence(self, sequence_type, lead):
        """Определение последовательности на основе типа лида"""
        sequences = {
            'demo_no_show': [
                {'delay': 0, 'action': 'email', 'template': 'demo_no_show_immediate'},
                {'delay': 1, 'action': 'task', 'type': 'call'},
                {'delay': 3, 'action': 'email', 'template': 'demo_reschedule'},
                {'delay': 7, 'action': 'linkedin', 'template': 'connection_request'},
                {'delay': 14, 'action': 'email', 'template': 'alternative_resources'}
            ],
            'proposal_sent': [
                {'delay': 3, 'action': 'email', 'template': 'proposal_follow_up'},
                {'delay': 7, 'action': 'task', 'type': 'call'},
                {'delay': 14, 'action': 'email', 'template': 'proposal_check_in'},
                {'delay': 21, 'action': 'task', 'type': 'executive_call'}
            ],
            'cold_outreach': [
                {'delay': 0, 'action': 'email', 'template': 'cold_intro'},
                {'delay': 4, 'action': 'linkedin', 'template': 'social_touch'},
                {'delay': 8, 'action': 'email', 'template': 'value_add_content'},
                {'delay': 15, 'action': 'task', 'type': 'call'},
                {'delay': 22, 'action': 'email', 'template': 'final_attempt'}
            ]
        }
        
        # Персонализация последовательности
        sequence = sequences[sequence_type]
        return self.personalize_sequence(sequence, lead)
    
    def schedule_action(self, lead_id, step):
        """Планирование конкретного действия"""
        execute_time = datetime.now() + timedelta(days=step['delay'])
        
        action_data = {
            'lead_id': lead_id,
            'action_type': step['action'],
            'template': step.get('template'),
            'execute_at': execute_time,
            'status': 'scheduled'
        }
        
        self.crm.create_task(action_data)
```

def calculate_lead_score(lead_data):
    score = 0
    
    # Demographic scoring
    if lead_data['company_size'] in target_company_sizes:
        score += 25
    if lead_data['industry'] in target_industries:
        score += 20
    if lead_data['title'] in target_titles:
        score += 15
    
    # Behavioral scoring
    score += min(lead_data['page_views'] * 2, 20)
    score += min(lead_data['time_on_site'] / 60 * 3, 15)
    score += lead_data['content_downloads'] * 5
    
    # Intent signals
    if lead_data['visited_pricing']:
        score += 15
    if lead_data['demo_request']:
        score += 25
    if lead_data['trial_signup']:
        score += 30
    
    return min(score, 100)
